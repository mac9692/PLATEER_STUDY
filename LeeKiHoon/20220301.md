## 코드
```java
import java.util.Scanner;

public class Main {
    
        public static void main(String[] args) {

        Scanner in = new Scanner(System.in);
        int a = in.nextInt();
        int b = in.nextInt();
        int c = in.nextInt();
        int reward = 0;

        if (a == b && b == c) {
            reward = 10000 + a * 1000;
        } else if (a == b) {
            reward = 1000 + a * 100;
        } else if (a == c) {
            reward = 1000 + a * 100;
        } else if (b == c) {
            reward = 1000 + b * 100;
        } else {
            if (a > b && a > c){
                reward = a * 100;
            } else if (b > a && b > c) {
                reward = b * 100;
            } else {
                reward = c * 100;
            }
        }

        System.out.println(reward);

    }
    
}
```
